# 核心契约 | Core Contracts

所有可插拔模块遵循统一契约；schema 版本用于追踪破坏性变更。
All pluggable modules follow unified contracts; schema version tracks breaking changes.

## SessionArtifact（萃取输入 | Distiller Input）

```typescript
interface SessionArtifact {
  schema_version: "1.0"
  meta: {
    session_id: string
    captured_at: string
    capture_trigger: string
    loam_version: string
    provider: string
  }
  context: {
    cwd: string
    worktree: string
    repo?: string
    branch?: string
    commit?: string
    dirty?: boolean
  }
  time_range: { start: string; end: string }
  session: Record<string, unknown>
  messages: Array<{
    id: string
    role: "user" | "assistant" | "system"
    timestamp: string
    content?: string
    parts?: ArtifactPart[]
  }>
  tools?: Array<{
    id: string
    message_id: string
    name: string
    input: Record<string, unknown>
    output?: string
    error?: string
  }>
  redacted: {
    patterns_applied: string[]
    redacted_count: number
  }
}

type ArtifactPart =
  | { type: "text"; text: string }
  | { type: "reasoning"; text: string }
  | { type: "tool"; name: string; input: unknown; output?: string; error?: string }
  | { type: "file"; filename: string; mime: string }
```

## DistillResultDraft（第三方返回值 | Draft Returned by Distiller）

```typescript
/**
 * distiller.run() 的返回类型。
 * 引擎接收后自动注入 id、fingerprint、distiller_id、distiller_version。
 * Return type of distiller.run().
 * Engine auto-injects id, fingerprint, distiller_id, distiller_version.
 */
type DistillResultDraft<T = Record<string, unknown>> = {
  type: string
  title: string
  summary: string
  confidence: number            // 0.0 ~ 1.0
  tags: string[]
  payload: T                    // 特定类型结构化数据，供 Sink 消费 | Structured payload for Sink mapping
  evidence: Array<{
    session_id: string
    message_id: string
    excerpt: string
    position?: { start: number; end: number }
    // trace_command 由引擎自动生成 | auto-generated by engine
  }>
  actions?: Array<{
    type: string
    label: string
    metadata?: Record<string, unknown>
  }>
  render?: {
    markdown?: string
    html?: string
  }
}
```

## DistillResult（框架完整类型 | Full Type, Engine-Managed）

```typescript
/**
 * 引擎在接收 DistillResultDraft 后注入元数据得到的完整结果。
 * 第三方无需构造此类型；Sink 消费此类型。
 * Full result after engine injects metadata from DistillResultDraft.
 * Third parties do NOT construct this; Sinks consume this.
 */
interface DistillResult<T = Record<string, unknown>> {
  id: string                    // 引擎注入 UUID v4 | Engine-injected UUID v4
  fingerprint: string           // 引擎注入 SHA-256 去重指纹 | Engine-injected dedup fingerprint
  distiller_id: string          // 引擎注入 | Engine-injected
  distiller_version: string     // 引擎注入 | Engine-injected
  type: string
  title: string
  summary: string
  confidence: number
  tags: string[]
  payload: T
  evidence: Array<{
    session_id: string
    message_id: string
    excerpt: string
    position?: { start: number; end: number }
    trace_command: string       // 引擎自动生成 | Engine auto-generated
  }>
  actions?: Array<{
    type: string
    label: string
    metadata?: Record<string, unknown>
  }>
  render?: { markdown?: string; html?: string }
}
```

## DistillerPlugin（萃取器接口 | Distiller Interface）

```typescript
/**
 * npm 包默认导出的工厂函数；接收配置，返回插件实例。
 * Factory function exported as default from a distiller npm package.
 */
type DistillerFactory = (config?: Record<string, unknown>) => DistillerPlugin

/**
 * 每个 distiller 插件（npm 包或本地模块）必须实现此接口。
 * Each distiller plugin (npm package or local module) must implement this interface.
 */
interface DistillerPlugin {
  id: string                   // 全局唯一，建议格式: '@scope/distiller-name' | Globally unique
  name: string
  version: string
  supported_types: string[]    // 该插件输出的 DistillResult.type 列表 | Result types this distiller produces
  configSchema?: JSONSchema7   // 声明式配置校验 schema | Declarative config validation schema
  payloadSchema?: Record<string, JSONSchema7>  // 按 type 映射 payload 结构，引擎传递给 Sink 用于运行时校验 | Per-type payload schema for runtime Sink validation

  // ---- 生命周期钩子 | Lifecycle hooks ----
  /** 引擎启动时调用一次，可做连接/预热 | Called once at engine startup */
  initialize?(ctx: DistillerContext): Promise<void>

  /** 核心萃取逻辑 | Core distillation logic */
  run(input: {
    artifactStore: ArtifactQueryClient   // 流式迭代器，避免全量内存加载 | Streaming iterator
    llm: LLMRouter
    state: DistillerStateKV              // 水位线 / 去重状态 | Watermark / dedup state
    config?: Record<string, unknown>
  }): Promise<DistillResultDraft[]>   // 返回 Draft；引擎自动注入 id/fingerprint/distiller_id | Return Draft; engine injects metadata

  /** 引擎停止时调用，释放资源 | Called at engine shutdown to release resources */
  teardown?(): Promise<void>
}

/** 插件初始化时注入的上下文 | Context injected during plugin initialization */
interface DistillerContext {
  dumpDir: string
  repo?: string
  logger: {
    info(msg: string): void
    warn(msg: string): void
    error(msg: string, err?: unknown): void
  }
}
```

## DistillEngine（萃取引擎 | Distill Engine）

```typescript
/**
 * 萃取引擎：从配置加载插件，协调调度各 distiller 运行。
 * Distill engine: loads plugins from config, orchestrates distiller execution.
 */
interface DistillEngine {
  /** 从 AICConfig 批量加载并初始化所有声明的 distillers | Load all distillers from config */
  loadFromConfig(config: AICConfig): Promise<void>

  /**
   * 执行萃取。不指定 distillers 则运行全部已注册插件。
   * Run distillation. If distillers omitted, runs all registered plugins.
   */
  run(options?: {
    distillers?: string[]   // 按 id 过滤；省略则运行全部 | Filter by id; omit to run all
    repo?: string
    since?: string          // ISO 8601
    until?: string          // ISO 8601
  }): Promise<DistillReport[]>
}

/** 单次 distiller 运行报告 | Single distiller run report */
interface DistillReport {
  distiller_id: string
  artifacts_processed: number
  results_produced: number
  results_skipped: number       // fingerprint 去重跳过 | Skipped due to dedup
  errors: Array<{ message: string; session_id?: string }>
  duration_ms: number
}
```

## AICConfig（配置文件结构 | Config File Shape）

```typescript
/**
 * loam.config.ts（或 aic.config.json）的根结构。
 * Root structure of loam.config.ts (or aic.config.json).
 */
interface AICConfig {
  dump_dir?: string              // 覆盖 LOAM_DUMP_DIR 环境变量 | Overrides LOAM_DUMP_DIR env

  capture?: {
    providers: string[]          // e.g., ["opencode"]
    debounce_ms?: number
  }

  /**
   * 声明要加载的 distiller 列表，引擎启动时按序动态 import。
   * Declare distillers to load; engine dynamically imports them in order at startup.
   *
   * - 字符串形式：npm 包名或相对路径 | String: npm package id or relative path
   * - 对象形式：可附带 per-distiller config | Object: with per-distiller config
   */
  distillers: Array<
    | string
    | { plugin: string; config: Record<string, unknown> }
  >

  sinks?: Array<
    | string
    | { plugin: string; config: Record<string, unknown> }
  >

  llm?: {
    default_budget?: "cheap" | "standard" | "premium"
    providers?: Record<string, {
      api_key?: string    // 推荐通过环境变量注入 | Prefer env var injection
      base_url?: string
    }>
  }
}
```

```typescript
  },
```


## @loamlog/distiller-sdk（第三方开发辅助 | Third-party Dev SDK）

第三方开发者应使用 `@loamlog/distiller-sdk`，而非直接实现原始接口。
Third parties should use `@loamlog/distiller-sdk` instead of implementing raw interfaces.

```typescript
import { defineDistiller, createEvidence } from '@loamlog/distiller-sdk'

/**
 * defineDistiller：高阶函数，自动处理样板代码。
 * - 自动注入 distiller_id、distiller_version 到 run 上下文
 * - 内置幂等状态管理（水位线自动更新）
 * defineDistiller: higher-order fn that removes boilerplate.
 * - Auto-injects distiller_id & distiller_version into run context
 * - Built-in idempotent state management (watermark auto-updated)
 */
function defineDistiller<TPayload = Record<string, unknown>>(
  spec: {
    id: string
    name: string
    version: string
    supported_types: string[]
    configSchema?: JSONSchema7
    payloadSchema?: Record<string, JSONSchema7>
    initialize?(ctx: DistillerContext): Promise<void>
    run(input: {
      artifactStore: ArtifactQueryClient
      llm: LLMRouter           // 不使用 LLM 可以安全忽略 | Safe to ignore if not using LLM
      state: DistillerStateKV  // 已验证命名空间隔离 | Namespace-isolated by engine
      config?: Record<string, unknown>
    }): Promise<DistillResultDraft<TPayload>[]>
    teardown?(): Promise<void>
  }
): DistillerPlugin

/**
 * createEvidence：快速构造证据条目，自动提取 session_id / message_id。
 * createEvidence: quickly build evidence item, auto-extracts session_id / message_id.
 */
function createEvidence(
  artifact: SessionArtifact,
  message: SessionArtifact['messages'][number],
  excerpt: string,
): DistillResultDraft['evidence'][number]
```

### 最小可用示例 | Minimal Custom Distiller Example

```typescript
// my-distiller/index.ts
import { defineDistiller, createEvidence } from '@loamlog/distiller-sdk'

export default defineDistiller({
  id: '@my-org/find-todos',
  name: 'TODO Extractor',
  version: '1.0.0',
  supported_types: ['todo-item'],

  async run({ artifactStore }) {
    const results = []

    for await (const artifact of artifactStore.getUnprocessed('@my-org/find-todos')) {
      for (const msg of artifact.messages) {
        if (msg.role === 'user' && msg.content?.includes('TODO:')) {
          results.push({
            type: 'todo-item',
            title: 'Found TODO in session',
            summary: (msg.content ?? '').slice(0, 80),
            confidence: 1.0,
            tags: ['todo'],
            payload: { raw_text: msg.content },
            evidence: [createEvidence(artifact, msg, msg.content ?? '')],
            // id / fingerprint / distiller_id 由引擎自动注入
          })
        }
      }
    }

    return results
  },
})
```
